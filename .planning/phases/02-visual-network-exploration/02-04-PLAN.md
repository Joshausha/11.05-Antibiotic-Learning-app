```typescript
type VisualizationMode = 'network' | 'filtered' | 'northwestern';

function NetworkVisualizationContainer({ pathogens, antibiotics, coverage }) {
  const [mode, setMode] = useState<VisualizationMode>('network');

  return (
    <div className="visualization-container">
      <TabBar>
        <Tab
          active={mode === 'network'}
          onClick={() => setMode('network')}
        >
          Interactive Network
        </Tab>
        <Tab
          active={mode === 'filtered'}
          onClick={() => setMode('filtered')}
        >
          Filter-Driven View
        </Tab>
        <Tab
          active={mode === 'northwestern'}
          onClick={() => setMode('northwestern')}
        >
          Northwestern Charts
        </Tab>
      </TabBar>

      <div className="visualization-content">
        {mode === 'network' && (
          <D3NetworkGraph
            pathogens={pathogens}
            antibiotics={antibiotics}
            coverage={coverage}
          />
        )}

        {mode === 'filtered' && (
          <FilterDrivenView
            pathogens={pathogens}
            antibiotics={antibiotics}
            coverage={coverage}
          />
        )}

        {mode === 'northwestern' && (
          <NorthwesternPieChart
            antibiotics={antibiotics}
          />
        )}
      </div>
    </div>
  );
}
```

**What to avoid and WHY:**

- Don't duplicate data → single source passed to all modes
