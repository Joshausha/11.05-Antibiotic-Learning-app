```typescript
interface TooltipProps {
  content: {
    name: string;
    type: 'pathogen' | 'antibiotic';
    gramStatus?: string;
    mechanism?: string;
    vulnerability?: string;
    coverageCount?: number;
  };
  position: { x: number; y: number };
}

function NetworkTooltip({ content, position }: TooltipProps) {
  return (
    <foreignObject
      x={position.x + 30}
      y={position.y - 20}
      width={250}
      height={120}
    >
      <div className="tooltip-container">
        <strong>{content.name}</strong>
        {content.type === 'pathogen' && (
          <>
            <div>Gram: {content.gramStatus}</div>
            <div>Vulnerability: {content.vulnerability}</div>
            <div>{content.coverageCount} effective antibiotics</div>
          </>
        )}
        {content.type === 'antibiotic' && (
          <>
            <div>Mechanism: {content.mechanism}</div>
            <div>Targets: {content.vulnerability}</div>
          </>
        )}
      </div>
    </foreignObject>
  );
}
```

**Update D3NetworkGraph.tsx with hover state:**
